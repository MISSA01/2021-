#import <Foundation/Foundation>;

@interface Chat : NSObject

//聊天消息的自增id
@property (nonatomic, strong) NSInteger chatId;
//消息发送人手机号
@property (nonatomic, assign) NSString fromPhoneNum;
//消息接收人手机号
@property (nonatomic, assign) NSString toPhoneNum;
//发送时间
@property (nonatomic, assign) NSString sentTime;
//消息内容
@property (nonatomic, assign) NSString chatContent;
//消息状态（1-已读，0-未读）
@property (nonatomic, strong) NSInteger chatState;
//消息类型（1-文字，0-图片，2是二维码）
@property (nonatomic, strong) NSInteger chatType;
//订单id
@property (nonatomic, strong) NSInteger orderId;


@end
