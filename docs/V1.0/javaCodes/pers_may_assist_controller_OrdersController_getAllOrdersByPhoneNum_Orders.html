public class Orders implements Serializable{

    private int orderId; //自增主键
    private int taskId; //对应任务ID
    private String phoneNum; //接任务的人的电话
    private String receiveTime; //接受时间
    private String finishTime; //完成时间
    private Task task; //本订单对应的任务的具体属性

    public int getOrderId(){
        return orderId;
    }

    public void setOrderId(int orderId){
        this.orderId = orderId;
    }

    public int getTaskId(){
        return taskId;
    }

    public void setTaskId(int taskId){
        this.taskId = taskId;
    }

    public String getPhoneNum(){
        return phoneNum;
    }

    public void setPhoneNum(String phoneNum){
        this.phoneNum = phoneNum;
    }

    public String getReceiveTime(){
        return receiveTime;
    }

    public void setReceiveTime(String receiveTime){
        this.receiveTime = receiveTime;
    }

    public String getFinishTime(){
        return finishTime;
    }

    public void setFinishTime(String finishTime){
        this.finishTime = finishTime;
    }

    public Task getTask(){
        return task;
    }

    public void setTask(Task task){
        this.task = task;
    }

}

public class Task implements Serializable{

    private int taskId; //自增主键
    private String phoneNum; //任务发起者电话
    private String taskTitle; //任务标题
    private int taskType; //任务类型
    private float taskReward; //任务奖励
    private String taskContent; //任务内容
    private String taskInTime; //任务发布时间
    private String taskOutTime; //任务截止时间
    private int taskState; //任务状态
    private int taskPeoNum; //任务人数
    private int taskAvaPeoNum; //任务人数

    public int getTaskId(){
        return taskId;
    }

    public void setTaskId(int taskId){
        this.taskId = taskId;
    }

    public String getPhoneNum(){
        return phoneNum;
    }

    }

    public String getTaskTitle(){

    }

    public void setTaskTitle(String taskTitle){
        this.taskTitle = taskTitle;
    }

    public int getTaskType(){
        return taskType;
    }

    public void setTaskType(int taskType){
        this.taskType = taskType;
    }

    public float getTaskReward(){
        return taskReward;
    }

    public void setTaskReward(float taskReward){
        this.taskReward = taskReward;
    }

    public String getTaskContent(){
        return taskContent;
    }

    public void setTaskContent(String taskContent){

    }

    public String getTaskInTime(){

    }

    public void setTaskInTime(String taskInTime){

    }

    public String getTaskOutTime(){

    public void setTaskOutTime(String taskOutTime){

    }

    public int getTaskState(){

    public void setTaskState(int taskState){

    public int getTaskPeoNum(){

        return taskAvaPeoNum;

        this.taskAvaPeoNum = taskAvaPeoNum;
    }

}
