public class Chat implements Serializable{

    private int chatId; //聊天消息的自增id
    private String fromPhoneNum; //消息发送人手机号
    private String toPhoneNum; //消息接收人手机号
    private String sentTime; //发送时间
    private String chatContent; //消息内容
    private int chatState; //消息状态（1-已读，0-未读）
    private int chatType; //消息类型（1-文字，0-图片，2是二维码）
    private int orderId; //订单id

    public int getChatId(){
        return chatId;
    }

    public void setChatId(int chatId){
        this.chatId = chatId;
    }

    public String getFromPhoneNum(){
        return fromPhoneNum;
    }

    public void setFromPhoneNum(String fromPhoneNum){
        this.fromPhoneNum = fromPhoneNum;
    }

    public String getToPhoneNum(){
        return toPhoneNum;
    }

    public void setToPhoneNum(String toPhoneNum){
        this.toPhoneNum = toPhoneNum;
    }

    public String getSentTime(){
        return sentTime;
    }

    public void setSentTime(String sentTime){
        this.sentTime = sentTime;
    }

    public String getChatContent(){
        return chatContent;
    }

    public void setChatContent(String chatContent){
        this.chatContent = chatContent;
    }

    public int getChatState(){
        return chatState;
    }

    public void setChatState(int chatState){
        this.chatState = chatState;
    }

    public int getChatType(){
        return chatType;
    }

    public void setChatType(int chatType){
        this.chatType = chatType;
    }

    public int getOrderId(){
        return orderId;
    }

    public void setOrderId(int orderId){
        this.orderId = orderId;
    }

}
