#import <Foundation/Foundation>;

@interface User : NSObject

//电话号码
@property (nonatomic, assign) NSString phoneNum;
//用户名
@property (nonatomic, assign) NSString userName;
//学号
@property (nonatomic, assign) NSString studentId;
//宿舍号
@property (nonatomic, strong) NSInteger dormNum;
//性别（1-男，0-女）
@property (nonatomic, strong) NSInteger gender;
//信誉分（满分10）
@property (nonatomic, strong) NSInteger userScore;
//头像URL
@property (nonatomic, assign) NSString avatar;


@end
